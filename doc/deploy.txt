def extract_connection_instances(plan_path):

    instances = {}

    f = open(plan_path, 'r')

    for line in f:

        if 'connection-instance/' in line and 'jndi-name="eis/' in line:

            start = line.find('jndi-name="') + 11
            end = line.find('"', start)

            jndi = line[start:end]

            instances[jndi] = 1

    f.close()

    return sorted(instances.keys())
